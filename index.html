function switchLang(lang, buttonEl) { // MAJ boutons actifs document.querySelectorAll('.lang-btn').forEach(btn => btn.classList.remove('active')); // Si la fonction est appelée par un clic, on active le bon bouton if (buttonEl) { buttonEl.classList.add('active'); } else { // Au démarrage, on cherche le bouton EN pour l'activer const defaultBtn = Array.from(document.querySelectorAll('.lang-btn')).find(btn => btn.innerText === lang.toUpperCase()); if (defaultBtn) defaultBtn.classList.add('active'); } // Textes statiques document.getElementById('hero-title').innerHTML = data[lang].heroTitle; document.getElementById('hero-desc').innerText = data[lang].heroDesc; document.querySelector('.lbl-pc-card').innerText = data[lang].cardPc; document.querySelector('.lbl-setup-card').innerText = data[lang].cardSetup; document.querySelector('.lbl-total-card').innerText = data[lang].cardTotal; document.getElementById('footer-text').innerText = data[lang].footer; // Headers Tableaux document.querySelector('.th-comp') ? document.querySelector('.th-comp').innerText = data[lang].thComp : null; document.querySelector('.th-item') ? document.querySelector('.th-item').innerText = data[lang].thItem : null; document.querySelectorAll('.th-src').forEach(el => el.innerText = data[lang].thSrc); document.querySelectorAll('.th-pr').forEach(el => el.innerText = data[lang].thPr); // Injection lignes PC const pcBody = document.querySelector('#table-pc tbody'); pcBody.innerHTML = ""; let currentCat = ""; data[lang].pcItems.forEach(item => { if(item.cat !== currentCat) { currentCat = item.cat; pcBody.innerHTML += `${data[lang].categories[currentCat]}`; } pcBody.innerHTML += ` ${item.spec} ${item.src} ${item.price} `; }); // Injection lignes SETUP const setupBody = document.querySelector('#table-setup tbody'); setupBody.innerHTML = ""; currentCat = ""; data[lang].setupItems.forEach(item => { if(item.cat !== currentCat) { currentCat = item.cat; setupBody.innerHTML += `${data[lang].categories[currentCat]}`; } setupBody.innerHTML += ` ${item.spec} ${item.src} ${item.price} `; }); } // Lance l'anglais proprement dès que le script est lu switchLang('en');
